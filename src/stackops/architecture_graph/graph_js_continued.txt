function svgPointFromEvent(event) {
  const point = svg.createSVGPoint();
  point.x = event.clientX;
  point.y = event.clientY;
  const screenCtm = svg.getScreenCTM();
  if (!screenCtm) {
    return { x: point.x, y: point.y };
  }
  return point.matrixTransform(screenCtm.inverse());
}

function finishDrag(event, group, node) {
  if (!dragState || dragState.node !== node || dragState.pointerId !== event.pointerId) {
    return;
  }
  group.releasePointerCapture(event.pointerId);
  if (dragState.moved) {
    node.pinned = true;
    selectedId = node.id;
    suppressNextClick = true;
    paint();
  }
  dragState = null;
}

function resize() {
  const rect = svg.getBoundingClientRect();
  width = Math.max(640, Math.round(rect.width));
  height = Math.max(480, Math.round(rect.height));
  svg.setAttribute("viewBox", `0 0 ${width} ${height}`);
  seedPositions();
}

function seedPositions() {
  const radius = Math.min(width, height) * 0.34;
  for (const node of nodes) {
    const index = Math.max(0, groups.indexOf(node.group));
    const angle = (index / Math.max(1, groups.length)) * Math.PI * 2;
    const hash = [...node.id].reduce((sum, char) => sum + char.charCodeAt(0), 0);
    node.r = Math.max(13, Math.min(24, 11 + Math.sqrt(node.fileCount + node.moduleCount) * 2.2));
    node.x = width / 2 + Math.cos(angle) * radius + (hash % 90) - 45;
    node.y = height / 2 + Math.sin(angle) * radius + ((hash * 7) % 90) - 45;
    node.vx = 0;
    node.vy = 0;
  }
}

function tick() {
  for (const edge of edges) {
    const dx = edge.targetNode.x - edge.sourceNode.x;
    const dy = edge.targetNode.y - edge.sourceNode.y;
    const distance = Math.max(1, Math.hypot(dx, dy));
    const desired = 168 + Math.min(96, Math.sqrt(edge.count) * 18);
    const force = (distance - desired) * 0.009;
    const fx = (dx / distance) * force;
    const fy = (dy / distance) * force;
    applyForce(edge.sourceNode, fx, fy);
    applyForce(edge.targetNode, -fx, -fy);
  }
  for (let index = 0; index < nodes.length; index += 1) {
    for (let otherIndex = index + 1; otherIndex < nodes.length; otherIndex += 1) {
      const left = nodes[index];
      const right = nodes[otherIndex];
      const dx = right.x - left.x;
      const dy = right.y - left.y;
      const distance = Math.max(1, Math.hypot(dx, dy));
      const minimum = left.r + right.r + 32;
      const strength = distance < minimum ? (minimum - distance) * 0.052 : 70 / (distance * distance);
      const fx = (dx / distance) * strength;
      const fy = (dy / distance) * strength;
      applyForce(left, -fx, -fy);
      applyForce(right, fx, fy);
    }
  }
  for (const node of nodes) {
    if (node.pinned) {
      continue;
    }
    const index = Math.max(0, groups.indexOf(node.group));
    const angle = (index / Math.max(1, groups.length)) * Math.PI * 2;
    const targetX = width / 2 + Math.cos(angle) * width * 0.24;
    const targetY = height / 2 + Math.sin(angle) * height * 0.22;
    node.vx += (targetX - node.x) * 0.006;
    node.vy += (targetY - node.y) * 0.006;
    node.vx *= 0.82;
    node.vy *= 0.82;
    node.x = Math.min(width - node.r - 12, Math.max(node.r + 12, node.x + node.vx));
    node.y = Math.min(height - node.r - 12, Math.max(node.r + 12, node.y + node.vy));
  }
}

function applyForce(node, fx, fy) {
  if (node.pinned) {
    return;
  }
  node.vx += fx;
  node.vy += fy;
}

function draw() {
  for (const { edge, line } of edgeViews) {
    const dx = edge.targetNode.x - edge.sourceNode.x;
    const dy = edge.targetNode.y - edge.sourceNode.y;
    const distance = Math.max(1, Math.hypot(dx, dy));
    const shorten = edge.targetNode.r + 5;
    const normalX = -dy / distance;
    const normalY = dx / distance;
    const offset = edge.parallelOffset;
    const startX = edge.sourceNode.x + normalX * offset;
    const startY = edge.sourceNode.y + normalY * offset;
    const endX = edge.targetNode.x + normalX * offset - (dx / distance) * shorten;
    const endY = edge.targetNode.y + normalY * offset - (dy / distance) * shorten;
    line.setAttribute("d", `M${startX},${startY} L${endX},${endY}`);
    line.setAttribute("stroke-width", Math.min(2.2, 0.45 + Math.sqrt(edge.count) * 0.18));
  }
  for (const { node, group, circle } of nodeViews) {
    group.setAttribute("transform", `translate(${node.x}, ${node.y})`);
    circle.setAttribute("r", node.r);
  }
}

function paint() {
  const query = search.value.trim().toLowerCase();
  const outgoing = new Set(edges.filter((edge) => edge.source === selectedId).map((edge) => edge.target));
  const incoming = new Set(edges.filter((edge) => edge.target === selectedId).map((edge) => edge.source));
  const focusedOutgoing = focusedModuleId ? new Set(edges.filter((edge) => edge.source === focusedModuleId).map((edge) => edge.target)) : new Set();
  const focusedIncoming = focusedModuleId ? new Set(edges.filter((edge) => edge.target === focusedModuleId).map((edge) => edge.source)) : new Set();
  const focusedNeighborhood = focusedModuleId ? new Set([focusedModuleId, ...focusedOutgoing, ...focusedIncoming]) : null;
  for (const { edge, line } of edgeViews) {
    const isOutgoing = edge.source === selectedId;
    const isIncoming = edge.target === selectedId;
    const isBidi = highlightBidirectional && bidiEdgeKeys.has(directedEdgeKey(edge.source, edge.target));
    const visibleByQuery = query === "" || edge.source.toLowerCase().includes(query) || edge.target.toLowerCase().includes(query);
    const visibleByFocus = !focusedNeighborhood || (focusedNeighborhood.has(edge.source) && focusedNeighborhood.has(edge.target));
    const hidden = !visibleByQuery || !visibleByFocus;
    line.setAttribute("class", ["edge", isOutgoing ? "outgoing" : "", isIncoming ? "incoming" : "", isBidi ? "bidi" : "", hidden ? "dim" : ""].join(" "));
    line.setAttribute("marker-end", isOutgoing ? "url(#arrow-out)" : isIncoming ? "url(#arrow-in)" : isBidi ? "url(#arrow-default)" : "url(#arrow-default)");
  }
  for (const { node, group } of nodeViews) {
    const connected = node.id === selectedId || outgoing.has(node.id) || incoming.has(node.id);
    const matches = query === "" || node.id.toLowerCase().includes(query) || node.label.toLowerCase().includes(query);
    const isBidiNode = highlightBidirectional && bidiNodeIds.has(node.id);
    const visibleByFocus = !focusedNeighborhood || focusedNeighborhood.has(node.id);
    const dim = (selectedId !== null && !connected) || !matches || !visibleByFocus;
    group.setAttribute("class", ["node", node.id === selectedId ? "selected" : "", isBidiNode ? "bidi" : "", dim ? "dim" : ""].join(" "));
  }
  renderDetails();
  renderBidiSummary();
}

function renderDetails() {
  const node = selectedId ? nodeById.get(selectedId) : null;
  detailTitle.textContent = node ? node.id : "Dependency Graph";
  detailMeta.textContent = node
    ? `${node.moduleCount} modules, ${node.fileCount} files, group ${node.group}`
    : `${payload.packageName}, depth ${payload.depth}, generated ${payload.generatedAt}`;
  renderEdgeList(outgoingList, edges.filter((edge) => edge.source === selectedId), "target");
  renderEdgeList(incomingList, edges.filter((edge) => edge.target === selectedId), "source");
}

function renderEdgeList(list, items, key) {
  list.replaceChildren();
  if (items.length === 0) {
    const empty = document.createElement("li");
    empty.className = "empty";
    empty.textContent = "None";
    list.append(empty);
    return;
  }
  for (const item of items.sort((left, right) => right.count - left.count || left[key].localeCompare(right[key]))) {
    const row = document.createElement("li");
    const name = document.createElement("b");
    const count = document.createElement("em");
    name.textContent = item[key];
    count.textContent = String(item.count);
    row.append(name, count);
    list.append(row);
  }
}

function renderBidiSummary() {
  if (!highlightBidirectional) {
    bidiSummarySection.hidden = true;
    return;
  }
  bidiSummarySection.hidden = false;
  bidiSummary.replaceChildren();
  const bidiPairs = [];
  const seen = new Set();
  for (const edge of edges) {
    const key = directedEdgeKey(edge.source, edge.target);
    if (!bidiEdgeKeys.has(key) || seen.has(key)) {
      continue;
    }
    const reverseKey = directedEdgeKey(edge.target, edge.source);
    seen.add(key);
    seen.add(reverseKey);
    const reverseEdge = edges.find((e) => directedEdgeKey(e.source, e.target) === reverseKey);
    bidiPairs.push({
      source: edge.source,
      target: edge.target,
      forwardCount: edge.count,
      reverseCount: reverseEdge ? reverseEdge.count : 0,
    });
  }
  if (bidiPairs.length === 0) {
    const emptyRow = bidiSummary.insertRow();
    const emptyCell = emptyRow.insertCell();
    emptyCell.colSpan = 3;
    emptyCell.textContent = "None";
    emptyCell.style.color = "#94a3b8";
    return;
  }
  const headerRow = bidiSummary.createTHead().insertRow();
  for (const text of ["Module A", "Module B", "A\u2192B", "B\u2192A"]) {
    const th = document.createElement("th");
    th.textContent = text;
    headerRow.append(th);
  }
  const tbody = bidiSummary.createTBody();
  for (const pair of bidiPairs.sort((left, right) => (right.forwardCount + right.reverseCount) - (left.forwardCount + left.reverseCount))) {
    const row = tbody.insertRow();
    const cellA = row.insertCell();
    cellA.textContent = pair.source;
    const cellB = row.insertCell();
    cellB.textContent = pair.target;
    const cellForward = row.insertCell();
    cellForward.textContent = String(pair.forwardCount);
    cellForward.className = "count";
    const cellReverse = row.insertCell();
    cellReverse.textContent = String(pair.reverseCount);
    cellReverse.className = "count";
  }
}

function renderLegend() {
  legend.replaceChildren();
  for (const group of groups) {
    const item = document.createElement("span");
    const swatch = document.createElement("i");
    swatch.style.background = groupColor.get(group);
    item.append(swatch, document.createTextNode(group));
    legend.append(item);
  }
}

function animate() {
  tick();
  draw();
  window.requestAnimationFrame(animate);
}
